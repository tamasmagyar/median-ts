# Typescript usage
```
import { median } from 'median';

const numbers: number[] = [1, 3, 3, 6, 7, 8, 9];
console.log(`The median is: ${median(numbers)}`);
```


# Javascript usage
```
const median = require('median-ts').median;

const numbers = [3, 1, 4, 1, 5, 9, 2, 6, 5];
console.log(median(numbers));
```
